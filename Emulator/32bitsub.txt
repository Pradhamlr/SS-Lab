DATA SEGMENT
    N1_LOW  DW 5678H
    N1_HIGH DW 1234H
    N2_LOW  DW 4321H
    N2_HIGH DW 8765H
    DIFF_LOW  DW ?
    DIFF_HIGH DW ?
    BORROW    DB ?
DATA ENDS

CODE SEGMENT
    ASSUME CS:CODE, DS:DATA

START:
    MOV AX, DATA
    MOV DS, AX

    ; --- Subtract low 16 bits ---
    MOV AX, N1_LOW
    SUB AX, N2_LOW        ; AX = low part difference
    MOV DIFF_LOW, AX
    MOV BL, 0
    JNC NO_BORROW         ; If no borrow, skip
    MOV BL, 1
NO_BORROW:

    ; --- Subtract high 16 bits with borrow ---
    MOV AX, N1_HIGH
    SBB AX, N2_HIGH       ; Subtract with borrow
    MOV DIFF_HIGH, AX
    MOV BORROW, BL

    MOV AH, 4CH
    INT 21H
CODE ENDS
