DATA SEGMENT
    ARR DW 1234H, 1200H, 2000H, 1000H, 3000H   ; hardcoded array
    LEN DW 5
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
	MOV AX, DATA
	MOV DS, AX
	
	MOV CX, LEN
	DEC CX
OUTER_LOOP:
	LEA SI, ARR
	MOV DX, CX
INNER_LOOP:
	MOV AX, [SI]
	CMP AX, [SI+2]
	JBE SKIP_SWAP
	
	XCHG AX, [SI+2]
	MOV [SI], AX
SKIP_SWAP:
	ADD SI, 2
	DEC DX
	JNZ INNER_LOOP
	
	LOOP OUTER_LOOP
	
	MOV AH, 4CH
	INT 21H
CODE ENDS
END START
