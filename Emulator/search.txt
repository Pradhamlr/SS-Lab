DATA SEGMENT
	ARR DW 1234H, 2345H, 4567H, 3456H
	LEN DW 5
	KEY DW 4345H
	MSG1 DB 10, 13, 'Number Found$'
	MSG2 DB 10, 13, 'Number Not Found$'
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
	MOV AX, DATA
	MOV DS, AX
	
	MOV CX, LEN
	LEA SI, ARR
	MOV AX, KEY
SEARCH_LOOP:
	CMP [SI], AX
	JE EQUAL
	
	ADD SI, 2
	LOOP SEARCH_LOOP
	
	LEA DX, MSG2
	MOV AH, 09H
	INT 21H
	JMP EXIT
EQUAL:
	LEA DX, MSG1
	MOV AH, 09H
	INT 21H
EXIT:
	MOV AH, 4CH
	INT 21H
CODE ENDS
END START
